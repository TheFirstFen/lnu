package com.calculator;


import java.util.List;

public class Calculate {
    private static Integer tempIndex;
    private static Double sum;

    public static Double calculate(List<String> input) {
        tempIndex = 0;
        sum = Double.parseDouble(input.get(tempIndex));
        for (String c : input) {
            if (c.equals("*")) {
                Double num2 = Double.parseDouble(input.get(tempIndex + 1));
                sum = Multiplication.multiplication(sum, num2);
            } else if (c.equals("/")) {
                Double num2 = Double.parseDouble(input.get(tempIndex + 1));
                sum = Division.division(sum, num2);
            } else if (c.equals("+")) {
                Double num2 = Double.parseDouble(input.get(tempIndex + 1));
                sum = Addition.addition(sum, num2);
            } else if (c.equals("-")) {
                Double num2 = Double.parseDouble(input.get(tempIndex + 1));
                sum = Subtract.subtract(sum, num2);
            } 
             tempIndex += 1;
        }
        return sum;
    }
}
